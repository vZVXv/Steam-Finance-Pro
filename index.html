Финансовый отчет Steam

Инструкция по получению данных

  1. Откройте страницу: История покупок Steam
  2. Авторизуйтесь в системе.
  3. Нажимайте кнопку "Загрузить больше операций" внизу страницы, пока не прогрузите историю до самого начала (это важно!).
  4. Нажмите F12 (или ПКМ -> Исследовать элемент) и перейдите во вкладку Console.
  5. Скопируйте скрипт ниже, вставьте в консоль и нажмите Enter.
  6. Вам скопируется вся информация о покупках в буфер обмена, создайте текстовый файл и вставьте информацию туда (CTRL+V).
  7. Замените расширение файла .txt на .JSON
(() => { const rows = document.querySelectorAll('tr.wallet_table_row'); const parseCurrency = (text) => { if (!text || text.trim() === "") return null; const clean = text.trim().replace(/\s+/g, ' '); const parts = clean.split(' '); let amount = parts[0].replace(/[^\d,.-]/g, '').replace(',', '.'); amount = amount.startsWith('+') ? amount.slice(1) : amount; const currency = parts.slice(1).join(' '); return { amount, currency }; }; const data = Array.from(rows).map(row => { const date = row.querySelector('.wht_date')?.innerText.trim(); const onclick = row.getAttribute('onclick') || ""; const transactionLink = onclick.match(/'([^']+)'/)?.[1] || ""; const type = row.querySelector('.wht_type div')?.innerText.trim(); const paymentMethod = row.querySelector('.wth_payment')?.innerText.trim().replace(/\s+/g, ' '); const itemsCell = row.querySelector('.wht_items'); let items; if (itemsCell?.innerText.includes('Торговая площадка Steam')) { items = "Торговая площадка Steam"; } else { const itemsList = itemsCell?.innerText.split('\n') .map(i => i.trim()) .filter(i => i && !i.includes('Подарок отправлен') && !i.includes('Получить помощь')); // Если предмет один — строка, если много — массив items = itemsList.length === 1 ? itemsList[0] : itemsList; } return { date, transaction_link: transactionLink, type, payment_method: paymentMethod, items: items, total: parseCurrency(row.querySelector('.wht_total')?.innerText), wallet_change: parseCurrency(row.querySelector('.wht_wallet_change')?.innerText), wallet_balance: parseCurrency(row.querySelector('.wht_wallet_balance')?.innerText) }; }); const json = JSON.stringify(data, null, 4); copy(json); return data; })();
